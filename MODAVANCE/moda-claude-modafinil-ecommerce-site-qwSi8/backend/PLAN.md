## Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

// в”Ђв”Ђв”Ђ ENUMS в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

enum Brand {
  SUN_PHARMA  // "Sun Pharma"
  HAB_PHARMA  // "HAB Pharma"
}

enum Category {
  MODAFINIL
  ARMODAFINIL
}

enum ProductBadge {
  BESTSELLER
  NEW
  SALE
  POPULAR
}

enum OrderStatus {
  PENDING
  CONFIRMED
  PROCESSING
  SHIPPED
  DELIVERED
  CANCELLED
}

enum PaymentMethod {
  BITCOIN
  ETHEREUM
  ZELLE
  BILL
}

enum CouponType {
  PERCENT
  FIXED
}

// в”Ђв”Ђв”Ђ USER в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model User {
  id           String   @id @default(cuid())
  email        String   @unique
  firstName    String
  lastName     String
  passwordHash String
  loyaltyPoints Int     @default(0)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  // Optional saved address (stored as JSON)
  savedAddress Json?

  orders       Order[]
  reviews      Review[]
}

// в”Ђв”Ђв”Ђ PRODUCT в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Product {
  id               String        @id @default(cuid())
  name             String
  slug             String        @unique
  brand            Brand
  category         Category
  strength         String        // "200mg"
  pillsPerStrip    Int
  description      String        @db.LongText
  shortDescription String        @db.Text
  image            String        // primary image URL (= images[0])
  images           Json          // String[] stored as JSON вЂ” MySQL has no native array type
  effects          Json          // String[] stored as JSON
  ingredients      String        @db.Text
  manufacturer     String
  rating           Float         @default(0)
  reviewCount      Int           @default(0)
  badge            ProductBadge?
  inStock          Boolean       @default(true)
  featured         Boolean       @default(false)
  createdAt        DateTime      @default(now())
  updatedAt        DateTime      @updatedAt

  variants         ProductVariant[]
  reviews          Review[]
  orderItems       OrderItem[]
}

model ProductVariant {
  id            String   @id @default(cuid())
  productId     String
  quantity      Int      // number of pills
  price         Float
  originalPrice Float?
  savings       Float?
  label         String?
  sortOrder     Int      @default(0)

  product       Product  @relation(fields: [productId], references: [id], onDelete: Cascade)
  orderItems    OrderItem[]
}

// в”Ђв”Ђв”Ђ ORDER в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Order {
  id                String        @id
  // ID format: ORD-{timestamp}-{random5}  e.g. ORD-1718123456789-ABC12
  // Generated in service, not by DB

  userId            String?       // null for guest checkout
  status            OrderStatus   @default(CONFIRMED)
  subtotal          Float
  shipping          Float         // 0 if subtotal >= 150, else 9.99
  discount          Float         @default(0)
  total             Float
  paymentMethod     PaymentMethod
  trackingNumber    String?
  estimatedDelivery DateTime?
  shippingAddress   Json          // Address object snapshot
  couponCode        String?       // for reference/reporting
  createdAt         DateTime      @default(now())
  updatedAt         DateTime      @updatedAt

  user              User?         @relation(fields: [userId], references: [id])
  items             OrderItem[]
  statusLogs        OrderStatusLog[]
}

model OrderItem {
  id          String  @id @default(cuid())
  orderId     String
  productId   String
  variantId   String
  productName String  // snapshot at order time
  quantity    Int     // number of packs
  pillCount   Int     // variant.quantity * quantity
  price       Float   // variant.price * quantity
  image       String  // product.image snapshot

  order       Order          @relation(fields: [orderId], references: [id], onDelete: Cascade)
  product     Product        @relation(fields: [productId], references: [id])
  variant     ProductVariant @relation(fields: [variantId], references: [id])
}

model OrderStatusLog {
  id         String      @id @default(cuid())
  orderId    String
  fromStatus OrderStatus?  // null = initial creation
  toStatus   OrderStatus
  note       String?
  changedAt  DateTime    @default(now())

  order      Order       @relation(fields: [orderId], references: [id], onDelete: Cascade)
}

// в”Ђв”Ђв”Ђ REVIEW в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Review {
  id        String   @id @default(cuid())
  productId String
  userId    String?  // null for seeded reviews
  userName  String
  rating    Int      // 1вЂ“5
  title     String
  body      String   @db.LongText
  verified  Boolean  @default(false)
  createdAt DateTime @default(now())

  product   Product  @relation(fields: [productId], references: [id], onDelete: Cascade)
  user      User?    @relation(fields: [userId], references: [id])
}

// в”Ђв”Ђв”Ђ COUPON в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Coupon {
  code      String     @id  // uppercase, e.g., "WELCOME10"
  discount  Float          // amount or percentage
  type      CouponType
  minOrder  Float      @default(0)
  expiresAt DateTime
  createdAt DateTime   @default(now())
}

// в”Ђв”Ђв”Ђ BLOG в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model BlogPost {
  id        String   @id @default(cuid())
  slug      String   @unique
  title     String
  excerpt   String
  body      String   @db.LongText
  image     String
  author    String
  category  String
  readTime  Int      // minutes
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// в”Ђв”Ђв”Ђ FAQ в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model FaqItem {
  id        String   @id @default(cuid())
  section   String
  question  String
  answer    String   @db.LongText
  sortOrder Int      @default(0)
  createdAt DateTime @default(now())
```

## Business Logic

### Order Creation
```
1. Validate all productIds + variantIds exist in DB
2. Calculate shipping: subtotal >= 150 в†’ 0, else 9.99
3. Validate coupon (if provided):
   - Exists in DB
   - Not expired (expiresAt > now)
   - Minimum order met (subtotal >= minOrder)
4. Calculate discount:
   - Coupon discount: percent в†’ subtotal * (discount/100), fixed в†’ discount value
   - Payment discount: bitcoin/ethereum в†’ 15%, zelle в†’ 10%, bill в†’ 0%
   - Both stack: totalDiscount = couponDiscount + paymentDiscount
5. total = subtotal - totalDiscount + shipping
6. Generate order ID: ORD-{Date.now()}-{randomUppercase5}
7. Create Order + OrderItems atomically in one Prisma transaction
8. Create initial OrderStatusLog entry (fromStatus: null, toStatus: CONFIRMED)
9. Return full order with items
```

### Order Status Transitions
```
Valid transitions:
PENDING    в†’ CONFIRMED, CANCELLED
CONFIRMED  в†’ PROCESSING, CANCELLED
PROCESSING в†’ SHIPPED, CANCELLED
SHIPPED    в†’ DELIVERED, CANCELLED
DELIVERED  в†’ (terminal вЂ” no changes allowed)
CANCELLED  в†’ (terminal вЂ” no changes allowed)
```

### Coupon Validation
```
POST /coupons/validate { code, subtotal }
1. Find coupon by code (case-insensitive в†’ toUpperCase before lookup)
2. If not found в†’ 404
3. If expiresAt < now в†’ 400 "Coupon has expired"
4. If subtotal < minOrder в†’ 400 "Minimum order $X required"
5. Return { valid: true, discount, type, discountAmount }
```

### Password Hashing
```
Register: passwordHash = await bcrypt.hash(password, 12)
Login:    valid = await bcrypt.compare(password, user.passwordHash)
```

### JWT Strategy
```
User JWT:
  payload: { sub: user.id, email: user.email, type: 'user' }
  secret: process.env.JWT_SECRET
  expiresIn: '7d'

Admin JWT:
  payload: { sub: admin.id, email: admin.email, type: 'admin' }
  secret: process.env.ADMIN_JWT_SECRET  (different secret!)
  expiresIn: '8h'
```

## Frontend Integration Notes

When migrating frontend from IndexedDB to this API:

1. Replace all `db.*` Dexie calls with `fetch('/api/...')` or axios
2. Replace `authStore` login/register with JWT API calls вЂ” store token in localStorage
3. Replace `adminStore` with admin JWT token
4. `cartStore` stays client-side (no cart persistence on backend needed)
5. `modavance-recently-viewed` localStorage stays client-side
6. `modavance-orders-seen-at` localStorage stays client-side (notification bell)
7. All React Query `queryFn` functions need to call the API instead of Dexie
8. Update CORS: frontend URL must be in backend `FRONTEND_URL` env var
9. API base URL in frontend: `VITE_API_URL=http://localhost:3000`
